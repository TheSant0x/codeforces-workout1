#include <bits/stdc++.h>
typedef long long ll;
using namespace std;
 void solve()
{
    int n;
    cin >> n;
    vector<int> a(n);
    for (auto &x : a)
        cin >> x;
    sort(a.begin(), a.end());
    int k = a[0];
    if (k % 2)
    {
        for (int i = 1; i < n; i++)
        {
            if (a[i] % 2 == 0 && (a[i] - k) % 2 == 0)
            {
                cout << "NO\n";
                return;
            }
        }
    }
    else
    {
        for (int i = 1; i < n; i++)
        {
            if (a[i] % 2 && (a[i] - k) % 2)
            {
                cout << "NO\n";
                return;
            }
        }
    }
    cout << "YES\n";
}
 int main()
{
    int t;
    cin >> t;
    while (t--)
        solve();
     return 0;
